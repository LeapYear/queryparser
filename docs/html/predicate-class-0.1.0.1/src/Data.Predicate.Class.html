module Data.Predicate.Class where
import Prelude hiding ((&&), (||), not)
import qualified Prelude
class Predicate p where
  infixr 3 &&
  (&&) :: p -> p -> p

  infixr 3 ||
  (||) :: p -> p -> p

  not :: p -> p

instance Predicate Bool where
  (&&) = (Prelude.&&)
  (||) = (Prelude.||)
  not = Prelude.not

instance (Predicate p) => Predicate (a -> p) where
  p && q = \ a -> p a && q a
  p || q = \ a -> p a || q a
  not p = not . p
